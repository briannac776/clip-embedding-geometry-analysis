\begin{lstlisting}[language=Python]
!pip install git+https://github.com/openai/CLIP.git
import torch
import clip
from PIL import Image
import os
import numpy as np
device = "cuda" if torch.cuda.is_available() else "cpu"
model, preprocess = clip.load("ViT-B/32", device=device)

import torchvision
import torchvision.transforms as transforms

transform = transforms.ToTensor()

dataset = torchvision.datasets.CIFAR10(
    root="./data",
    train=False,
    download=True,
    transform=transform
)

target_labels = {
    "cat": 3,
    "dog": 5,
    "automobile": 1,
    "truck": 9
}

from PIL import Image
import numpy as np

selected_images = []
selected_labels = []
selected_names = []

counts = {label: 0 for label in target_labels.values()}

def tensor_to_pil(img_tensor):
    img = img_tensor.permute(1, 2, 0).numpy()
    img = (img * 255).astype(np.uint8)
    return Image.fromarray(img)

for img_tensor, label in dataset:
    if label in counts and counts[label] < 50:
        selected_images.append(tensor_to_pil(img_tensor))
        selected_labels.append(label)
        selected_names.append(label)
        counts[label] += 1
   
    if all(c == 50 for c in counts.values()):
        break
from collections import Counter
Counter(selected_labels)

label_names = {v: k for k, v in target_labels.items()}
selected_label_names = [label_names[l] for l in selected_labels]

import clip
import torch

device = "cuda" if torch.cuda.is_available() else "cpu"
model, preprocess = clip.load("ViT-B/32", device=device)

image_embeddings = []

for img in selected_images:
    image_input = preprocess(img).unsqueeze(0).to(device)
    with torch.no_grad():
        embedding = model.encode_image(image_input)
    image_embeddings.append(embedding.cpu().numpy())

import numpy as np
image_embeddings = np.vstack(image_embeddings)

import pandas as pd

pd.DataFrame(image_embeddings).to_csv("clip_embeddings.csv", index=False)
pd.DataFrame({"label": selected_label_names}).to_csv("labels.csv", index=False)

os.makedirs("images", exist_ok=True)
from collections import defaultdict

label_counters = defaultdict(int)

for img, label_name in zip(selected_images, selected_label_names):
    label_counters[label_name] += 1
    filename = f"{label_name}_{label_counters[label_name]:03d}.png"
    img.save(os.path.join("images", filename)
\end{lstlisting}

\subsection{Experiment III: Neighborhood Purity Analysis}

> embeddings <- as.matrix(read.csv("clip_embeddings.csv"))
> labels <- read.csv("labels.csv")$label
> euclidean_dist <- function(x, Y) {
+    sqrt(rowSums((Y - matrix(x, nrow(Y), length(x), byrow = TRUE))^2))
+ }
> 
> cosine_sim <- function(x, Y) {
+    x_norm <- sqrt(sum(x^2))
+    Y_norms <- sqrt(rowSums(Y^2))
+    (Y %*% x) / (Y_norms * x_norm)
+ }
> set.seed(123)

> k_values <- c(3, 5, 10)

+    cosine_scores <- c()

+        x <- embeddings[i, ]
+        label <- labels[i]
+        
+        # Euclidean
+        d <- euclidean_dist(x, embeddings)
+        d[i] <- Inf
+        euc_neighbors <- order(d)[1:k]

import umap
import numpy as np
import pandas as pd

embeddings = pd.read_csv("clip_embeddings.csv").values

reducer = umap.UMAP(n_neighbors=15, min_dist=0.1, random_state=123)
umap_2d = reducer.fit_transform(embeddings)

pd.DataFrame(umap_2d, columns=["UMAP1", "UMAP2"]).to_csv("umap_raw.csv", index=False)

emb_norm = embeddings / np.linalg.norm(embeddings, axis=1, keepdims=True)

umap_norm = reducer.fit_transform(emb_norm)
pd.DataFrame(umap_norm, columns=["UMAP1", "UMAP2"]).to_csv("umap_norm.csv", index=False)
